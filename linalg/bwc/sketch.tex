\pagestyle{empty}
\newcount\nj
\newcount\nt
\newcount\vj
\newcount\vt
\newdimen\dj
\newdimen\dt
\newdimen\Dx
\newdimen\Dy
\newdimen\tmp

\def\main{%
\divide\dj by \nj\relax
\dt=\dj\divide\dt by \nt\relax
\tmp=0pt
\vj=0\relax
\loop\ifnum\vj<\nj\relax
    \bob{\tmp}{linewidth=2pt}%
    {%
    \vt=0\loop\ifnum\vt<\nt\relax
        \global\advance\tmp\dt
        \bob{\tmp}{linewidth=1pt,linestyle=dotted}%
        \advance\vt by 1\relax
    \repeat
    \relax
    }%
    \advance\vj by 1\relax
\repeat
\relax
\bob{\tmp}{linewidth=2pt}%
}

\def\drawcut#1#2x#3,#4x#5{%
\Dx=#1\relax
\Dy=#1\relax
\begin{pspicture}(\Dx,\Dy)

\def\bob##1##2{\psline[##2](0,##1)(\Dx,##1)}
\nj=#2\nt=#4\dj=\Dy\main

\def\bob##1##2{\psline[##2](##1,0)(##1,\Dy)}
\nj=#3\nt=#5\dj=\Dy\main\dj=\Dx\main

\psline(0,\Dy)(\Dx,0)
\end{pspicture}
\bigskip

}
